# 快速开发

# 开发规范、准则

  1. 如果该sdk需要传入参数必须传入对象，并且该参数有不一致性时，需要进行校验、并且没有参数时，需要预制默认值， 不要在实现类型中添加
  2. 对于环境异常必须返回sdkEnvErrorResponse方法
    对于api错误处理必须返回sdkApiReturnErrorResponse方法
    对于api成功处理必须返回sdkApiReturnSuccessResponse方法
    以上方法均在utils/response文件中，方便定位，并且它们会直接生产响应输出返回

# 开发介绍

在counselor-sdk根目录执行 yarn run gen 目录名/sdk名称 会为你自动生产模版

例如：yarn run gen system/helloworld 那么将在modules目录下的system中生产helloworld文件夹

如果system不存在会自动创建该目录并创建index.js，执行以上操作后自动在index.js中注册，命令行中会
提示以下信息告知你相关信息

counselor-sdk: 模块生成完毕路径：/Users/xxx/Desktop/gitProject/counselor-sdk/lib/modules/system/helloworld
counselor-sdk: 开始向该模块入口文件注入
counselor-sdk: 模块路径注入入口文件成功

执行以上操作你会看到helloworld文件夹中会出现wisedu.js、wxwork.js、index.js文件

index.js如下

import { transformApiBindProps } from '../../../utils/transform'
import wxworkImpletement from './wxwork'
import wiseduImpletement from './wisedu'
import learningPowerImpletement from './learningPower'

const apiName = 'helloworld'

const apiSwitch = {
  wisedu: wiseduImpletement,
  wxwork: wxworkImpletement,
  micromessenger: wxworkImpletement,
  learningPower: learningPowerImpletement,
}

function innerHelloworld (platform, args) {
  return apiSwitch[platform](platform, apiName, args)
}

export const helloworld = (args) => {
  const featureFunction = transformApiBindProps(innerHelloworld, args)

  // transformApiBindProps内部判断了平台是否为企业微信或者今日校园等平台 如果不是返回false，静默失败
  if (featureFunction) {
    return featureFunction()
  }
}

utils.js如下（以下只是示例代码!!）

import {
  isPlainObject,
  validateFail,
} from '../../../utils/base'

/**
 * @description 校验参数设置默认值
 * @param {object} args  对象属性如下
 * @return {object} args
 */
export function checkArgs (apiName, args = {}) {
  const { needResult, keeping } = args

  if (isPlainObject(args)) {
    if (needResult && typeof needResult !== 'number') {
      validateFail(`${apiName}：参数中needResult必须为number类型`)
    }

    const needResultValues = [
      0,
      1,
    ]

    if (needResult && needResultValues.some((needResultValue) => needResultValue !== needResult)) {
      validateFail(`${apiName}：参数中needResult只能为${needResultValues.map((it) => it).join(',')}`)
    }

    if (keeping && typeof keeping !== 'boolean') {
      validateFail(`${apiName}：参数中keeping必须为boolean类型`)
    }
  }

  return {
    keeping: keeping || false,
    needResult: needResult || 1,
  }
}
