# 快速上手

# 项目中启动 sdk

以下示例为 vue-cli 默认创建的 main.js 文件

import Vue from "vue";
import App from "./App.vue";
import bootStrap from "counselor-sdk";

Vue.config.productionTip = false;
//该函数是了为counselor-sdk获取初始化微信sdk的参数
// bootStrap将在内部执行这个函数，该函数返回值应遵照
// {code ,data ,message}结构
function request() {
  // xxxxxxxxxx
}

async function createApp() {
  const initOptions = {};
  // 此方法细节见Api中
  const { code } = await bootStrap(request, initOptions);

  const vue = new Vue({
    render: (h) => h(App),
  }).$mount("#app");
}

createApp();

执行 bootStrap 会遵照约定返回{code, data, message}给你，当失败时，会返回对于的信息和code方便你在业务中进行处理 以上操作执行正常时会自动注入对应平台的sdk脚本文件！，并且控制台中会告知你相关进度状态

# Api使用

counselor-sdk将今日校园、微信平台的jssdk分成了4大类别，当然开发人员可以自行维护新的类别
分别为 location、view、system、request
以下使用system中的scanQRCode（扫码二维码为例）

  import { scanQRCode } from 'counselor-sdk'

  <script>
    export default {
      async mounted() {
        const { code, data } = await scanQRCode()

        if (code === 0) {
          // 成功
        } else {
          // 失败
        }
      }
    }
  </script>
